1. API 호출 — fetch 기본 패턴

const API = 'http://localhost:3000';  // 백엔드 주소

// ① 파일 업로드
async function uploadImage(file) {
  const form = new FormData();
  form.append('image', file);          // input[type=file].files[0]

  const res = await fetch(`${API}/upload`, { method: 'POST', body: form });
  const data = await res.json();
  return data.jobId;                   // 이후 모든 요청에 사용
}

// ② 스캔 요청
async function startScan(jobId, tool = 'both') {
  const res = await fetch(`${API}/scan`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ jobId, tool }),  // tool: 'trivy' | 'grype' | 'both'
  });
  return res.json();
}

// ③ 결과 조회 (폴링 — 1초마다 status 확인)
async function pollResult(jobId, onDone) {
  const timer = setInterval(async () => {
    const res  = await fetch(`${API}/scan/${jobId}`);
    const data = await res.json();

    if (data.status === 'done') {
      clearInterval(timer);
      onDone(data);          // 완료 시 data를 차트에 넘김
    } else if (data.status === 'failed') {
      clearInterval(timer);
      console.error('스캔 실패:', data);
    }
  }, 2000);
}

// ④ 잡 목록 (최근 20개)
async function listJobs() {
  const res = await fetch(`${API}/scan?limit=20`);
  return res.json();
}

// ── 심각도 집계 함수 ──────────────────────────────────
function countSeverity(results, severityKey = 'severity') {
  const counts = { CRITICAL: 0, HIGH: 0, MEDIUM: 0, LOW: 0, UNKNOWN: 0 };
  results.forEach(v => {
    const sev = (v[severityKey] || 'UNKNOWN').toUpperCase();
    // Grype는 'Critical'(대문자) 형태로 오므로 toUpperCase() 처리
    const key = counts.hasOwnProperty(sev) ? sev : 'UNKNOWN';
    counts[key]++;
  });
  return counts;
}

// ── Chart.js 도넛 생성 ────────────────────────────────
function createDonut(canvasId, counts) {
  const colors = {
    CRITICAL: '#dc2626',
    HIGH:     '#ea580c',
    MEDIUM:   '#ca8a04',
    LOW:      '#16a34a',
    UNKNOWN:  '#6b7280',
  };

  new Chart(document.getElementById(canvasId), {
    type: 'doughnut',
    data: {
      labels: Object.keys(counts),
      datasets: [{
        data:            Object.values(counts),
        backgroundColor: Object.keys(counts).map(k => colors[k]),
        borderWidth: 2,
      }],
    },
    options: {
      cutout: '65%',
      plugins: {
        legend: { position: 'right' },
        tooltip: {
          callbacks: {
            label: ctx => ` ${ctx.label}: ${ctx.raw}건`,
          },
        },
      },
    },
  });
}

4. KEV 등재 취약점 테이블

// CISA KEV(Known Exploited Vulnerabilities) 등재 항목만 필터
function renderKevTable(grypeResults, containerId) {
  const kev = grypeResults
    .filter(v => v.kev_included)
    .sort((a, b) => (b.epss_score || 0) - (a.epss_score || 0));  // EPSS 높은 순

  if (!kev.length) {
    document.getElementById(containerId).innerHTML = '<p>KEV 등재 취약점 없음</p>';
    return;
  }

  const rows = kev.map(v => `
    <tr>
      <td><span class="kev-badge">KEV</span> ${v.vulnerability_id}</td>
      <td>${v.pkg_name} ${v.installed_version}</td>
      <td class="sev-${v.severity.toLowerCase()}">${v.severity}</td>
      <td>${v.epss_score ? (v.epss_score * 100).toFixed(1) + '%' : '-'}</td>
      <td>${v.fix_versions || '패치 없음'}</td>
    </tr>
  `).join('');

  document.getElementById(containerId).innerHTML = `
    <table>
      <thead>
        <tr><th>CVE</th><th>패키지</th><th>심각도</th><th>EPSS</th><th>수정 버전</th></tr>
      </thead>
      <tbody>${rows}</tbody>
    </table>
  `;
}

5. Top 10 위험 취약점 (EPSS 높은 순)

function renderTopRisks(grypeResults, containerId, limit = 10) {
  const top = [...grypeResults]
    .sort((a, b) => (b.epss_score || 0) - (a.epss_score || 0))
    .slice(0, limit);

  const rows = top.map((v, i) => `
    <tr>
      <td>${i + 1}</td>
      <td>${v.vulnerability_id}</td>
      <td>${v.pkg_name}</td>
      <td class="sev-${v.severity.toLowerCase()}">${v.severity}</td>
      <td>${v.epss_score ? (v.epss_score * 100).toFixed(2) + '%' : '-'}</td>
      <td>${v.risk_score ?? '-'}</td>
      <td>${v.kev_included ? '<span class="kev-badge">KEV</span>' : ''}</td>
    </tr>
  `).join('');

  document.getElementById(containerId).innerHTML = `
    <table>
      <thead>
        <tr><th>#</th><th>CVE</th><th>패키지</th><th>심각도</th>
            <th>EPSS</th><th>위험도</th><th>KEV</th></tr>
      </thead>
      <tbody>${rows}</tbody>
    </table>
  `;
}

function renderSecrets(securityList, containerId) {
  if (!securityList.length) {
    document.getElementById(containerId).innerHTML = '<p>시크릿 탐지 없음 ✅</p>';
    return;
  }

  const rows = securityList.map(s => `
    <tr>
      <td class="sev-${s.severity.toLowerCase()}">${s.severity}</td>
      <td>${s.rule_id}</td>
      <td>${s.category || '-'}</td>
      <td>${s.target}</td>
      <td>${s.match_masked || '-'}</td>
    </tr>
  `).join('');

  document.getElementById(containerId).innerHTML = `
    <table>
      <thead>
        <tr><th>심각도</th><th>Rule ID</th><th>유형</th><th>파일</th><th>탐지 내용</th></tr>
      </thead>
      <tbody>${rows}</tbody>
    </table>
  `;
}

7. 전체 통합 — 스캔 후 모든 차트 한 번에 렌더링

const API = 'http://localhost:3000';

async function runFullScan(file) {
  // 1) 업로드
  const form = new FormData();
  form.append('image', file);
  const { jobId } = await fetch(`${API}/upload`, { method: 'POST', body: form }).then(r => r.json());

  // 2) 스캔 요청
  await fetch(`${API}/scan`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ jobId, tool: 'both' }),
  });

  // 3) 완료까지 폴링
  return new Promise((resolve, reject) => {
    const timer = setInterval(async () => {
      const data = await fetch(`${API}/scan/${jobId}`).then(r => r.json());
      if (data.status === 'done') { clearInterval(timer); resolve(data); }
      if (data.status === 'failed') { clearInterval(timer); reject(data); }
    }, 2000);
  });
}

// ── 사용 예 (파일 input 이벤트에서) ──────────────────────
document.getElementById('fileInput').addEventListener('change', async (e) => {
  const file = e.target.files[0];
  const data = await runFullScan(file);

  // 차트 렌더링
  createDonut('trivyDonut', countSeverity(data.trivy.results));
  createDonut('grypeDonut', countSeverity(data.grype.results));
  renderKevTable(data.grype.results, 'kevTable');
  renderTopRisks(data.grype.results, 'topRisks');
  renderSecrets(data.filtered.security, 'secretList');
});
